      Parts:
      • - LED: 1
        - Button: 2
        - Wires: 7
        - 220 Ohm resistors: 1
        - 10K Ohm resistors: 2 

                              
                            import time
                            from machine import Pin
                            
                            led = Pin(2, Pin.OUT)        
                            button = Pin(13, Pin.IN,Pin.PULL_UP)
                            button2 = Pin(12, Pin.IN,Pin.PULL_UP)
                            
                            def reverseGPIO():
                                if led.value(0):
                                    led.value(1)

                                    led.value(0)
